224/* instruction register bit positions, see HD44780U data sheet */
225#define LCD_CLR 0 /* DB0: clear display */
226#define LCD_HOME 1 /* DB1: return to home position */
227#define LCD_ENTRY_MODE 2 /* DB2: set entry mode */
228#define LCD_ENTRY_INC 1 /* DB1: 1=increment, 0=decrement */
229#define LCD_ENTRY_SHIFT 0 /* DB2: 1=display shift on */
230#define LCD_ON 3 /* DB3: turn lcd/cursor on */
231#define LCD_ON_DISPLAY 2 /* DB2: turn display on */
232#define LCD_ON_CURSOR 1 /* DB1: turn cursor on */
233#define LCD_ON_BLINK 0 /* DB0: blinking cursor ? */
234#define LCD_MOVE 4 /* DB4: move cursor/display */
235#define LCD_MOVE_DISP 3 /* DB3: move display (0-> cursor) ? */
236#define LCD_MOVE_RIGHT 2 /* DB2: move right (0-> left) ? */
237#define LCD_FUNCTION 5 /* DB5: function set */
